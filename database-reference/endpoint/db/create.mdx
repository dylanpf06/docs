---

sidebarTitle: "/v1/db/create"
api: "GET https://api.dscloud.pt/v1/db/create"
authMethod: "authorization"
description: "Create a new database on your Ds Cloud account."
requestBody:
  content:
    application/json:
      schema:
        type: object
        required:
        - name
        - region
        properties:
          name:
            type: string
            description: "The desired name for your database."
          region:
            type: string
            description: "The region where you want to create the database. Refer to our documentation for available regions."
          plan:
            type: string
            description: "The plan tier for your database. (Optional)"
          cluster:
            type: string
            description: "The cluster where you want to create the database. (Optional)"
  required: true
requestFields:
  - name: name
    type: string
    description: "The desired name for your database."
    required: true
  - name: region
    type: string
    description: "The region where you want to create the database. Refer to our documentation for available regions."
    required: true
  - name: plan
    type: string
    description: "The plan tier for your database. (Optional)"
  - name: cluster
    type: string
    description: "The cluster where you want to create the database. (Optional)"
authHeader:
  name: Authorization
  type: string
  placeholder: "API KEY"
  description: "The API key for your account. You can find this in your [account settings](https://dscloud.pt/dashboard/settings)."
response:
  content:
    application/json:
      schema:
        type: object
        properties:
          status:
            type: string
            description: "Indicates whether the call was successful. `success` if successful, `error` if not."
          message:
            type: string
            description: "Detailed message describing the outcome."
          data:
            type: object
            description: "Database details upon successful creation."
            properties:
              id:
                type: string
                description: "The unique identifier of the created database."
              name:
                type: string
                description: "The name of the created database."
              region:
                type: string
                description: "The region where the database is located."
              plan:
                type: string
                description: "The plan tier assigned to the database. (If provided in the request)"
              cluster:
                type: string
                description: "The cluster where the database is located. (If provided in the request)"
              created_at:
                type: string
                description: "The timestamp of the database creation."
responseExamples:
