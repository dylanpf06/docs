---
title: "List all databases"
sidebarTitle: "/v3/db/list"
api: "GET https://api.dscloud.pt/v3/db/list"
authMethod: "authorization"
description: "Retrieve a list of all databases associated with your account."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The API key for your account. You can find this in your [account settings](https://dscloud.pt/dashboard/settings).
</ParamField>

### Response

<ResponseField name="status" type="string">
  Indicates whether the call was successful. `success` if successful, `error` if not.
</ResponseField>

<ResponseField name="response" type="array">
  A list of databases associated with your account.

  <Expandable title="Database Details">
    <ResponseField name="id" type="string">
      The ID of the database.
    </ResponseField>

    <ResponseField name="name" type="string">
      The name of the database.
    </ResponseField>

    <ResponseField name="ip" type="string">
      The IP address assigned to the database.
    </ResponseField>

    <ResponseField name="type" type="string">
      The type of the database (e.g., `mysql`, `mongodb`).
    </ResponseField>

    <ResponseField name="created_at" type="string">
      The timestamp of when the database was created.
    </ResponseField>
  </Expandable>
</ResponseField>
