---
title: "Rotate Database Password"
sidebarTitle: "/v1/db/rotate-password"
api: "POST https://api.dscloud.pt/v1/db/rotate-password/{id}"
authMethod: "authorization"
description: "Generate a new password for the specified database. This will immediately invalidate the previous password."
---

<ParamField header="Authorization" type="string" placeholder="API KEY" required>
  The API key for your account. You can find this in your [account settings](https://dscloud.pt/dashboard/settings).
</ParamField>

<ParamField name="id" type="string" urlParam required>
  The unique identifier of the database to rotate the password for.
</ParamField>

### Response

<ResponseField name="status" type="string">
  Indicates whether the call was successful. `success` if successful, `error` if not.
</ResponseField>

<ResponseField name="response" type="array">
  Contains the new database credentials.

  <ResponseField name="id" type="string">
    The ID of the database that had its password rotated.
  </ResponseField>

  <ResponseField name="new_password" type="string">
    The newly generated password for the database. Store this securely as it will not be shown again.
  </ResponseField>
</ResponseField>

<ResponseExample>
  ```json Response
  {
    "status": "success",
    "response": {
      "id": "003dd41a",
      "new_password": "n3w!s3cur3P@ssw0rd"
    }
  }
  ```
</ResponseExample>
